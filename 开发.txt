create table zbx_object_group(
  id varchar(50),
  name varchar(100),
  mark varchar(500),
  code varchar(50),
  status varchar(50),
  sort decimal(5),
   dr        varchar(1)  null,
  create_time datetime    null,
  create_by   varchar(50) null,
  update_time datetime    null,
  update_by   varchar(50) null,
  primary key (id)
);
